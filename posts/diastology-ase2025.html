━━━━━━━━━━━━━━━━━━━━━━━ ▎2025 ASE 指南：精神不變，變數擴增 新增替代變數： ・Pulmonary vein S/D 逆轉 ・Pulmonary vein A 反流異常 ・IVRT ・LA reservoir strain ・L lump ・Pulmonic regurgitation end-diastolic velocity Dr. Hanna 加一項：嚴重 LV 肥厚（ESC HFA-PEFF 演算法用） ▎整體邏輯 ・2 項 → 舒張異常 + LA 壓力正常/不明 ・3 項 → 舒張異常 + LA 壓力升高 → E/A <2：Grade II → E/A ≥2：Grade III ━━━━━━━━━━━━━━━━━━━━━━━ ▎從二尖瓣血流一眼看出 LA 壓力 【綠框一：反向 E/A <0.8 + E <50 cm/s】 → LA 壓力正常 【綠框二：E/A >2 在 >40 歲患者】 → LA 壓力升高 ━━━━━━━━━━━━━━━━━━━━━━━ ▎E' 為什麼是最純粹的舒張功能指標 ▸ 2025 新正常值（多數病人 >40 歲） ・Medial E' ≥ 6 cm/s ・Lateral E' ≥ 7 cm/s（40–65 歲可放寬 8） ・平均 E' ≥ 6.5–7 cm/s ▸ E' 不可靠的四種心衰竭 ① AF ② 高輸出心衰 ③ LVEF 正常 MR ④ Constrictive pericarditis ▸ E' 數值不準：人工二尖瓣、瓣環成形環、嚴重 MAC ━━━━━━━━━━━━━━━━━━━━━━━ ▎E/E' 公式：LA 壓力最純粹的替代 E ∝ LA 壓力 × LV 放鬆 E' ∝ LV 放鬆 ∴ E/E' ∝ LA 壓力 【異常標準】 ・Septal E/E' ≥ 15 ・Lateral E/E' ≥ 13 ・Average E/E' ≥ 14 【重要計算】 Average E/E' = E ÷ [(Septal E' + Lateral E') / 2] ━━━━━━━━━━━━━━━━━━━━━━━ ▎LA 容積：LA 壓力的歷史足跡 LA 擴大反映某時刻壓力曾升高，不等於當下仍高。 ━━━━━━━━━━━━━━━━━━━━━━━ ▎E DT：高壓下 E 又高又短 【異常：DT <140–160 ms】 ━━━━━━━━━━━━━━━━━━━━━━━ ▎L lump 【異常：L wave ≥ 40 cm/s】 ━━━━━━━━━━━━━━━━━━━━━━━ ▎IVRT：反直覺 ・代償期舒張異常 → IVRT 延長 ・失代償高 LA 壓力 → IVRT <70 ms ━━━━━━━━━━━━━━━━━━━━━━━ ▎肺靜脈波形 S 對應 X descent（S=X） D 對應 Y descent A reversal 對應心房收縮 ▸ S/D 異常：S/D <2/3，嚴重者 S reversal ▸ 完整 S reversal：重度 MR 最特異 ▎PV A reversal（LVEDP 獨家指標） ・振幅 >35 cm/s ・比 mitral A duration 長 >30 ms ━━━━━━━━━━━━━━━━━━━━━━━ ▎B bump（M-mode）→ 反映 LVEDP ━━━━━━━━━━━━━━━━━━━━━━━ ▎實例：Grade III 舒張功能異常 Dr. Hanna 示範案例： ・Lateral E' = 6 cm/s ・E/A = 3 ・L lump ≈ 50 cm/s ・DT <160 ms ・Lateral E/E' = 23 ・PV S/D 反轉 ・PV A duration 157 ms vs. mitral A 95 ms ━━━━━━━━━━━━━━━━━━━━━━━ ▎LA Reservoir Strain 【正常：>+18%】 低 LA reservoir strain = LA 壓力升高 ━━━━━━━━━━━━━━━━━━━━━━━ ▎PR velocity ▸ End-diastolic velocity² ≈ PAD − RVEDP PAD ≈ velocity² + RA pressure ━━━━━━━━━━━━━━━━━━━━━━━ ▎特殊情境 ①｜AF ・Medial E/E' cutoff 改為 ≥ 11 ・推薦：低 DT、低 IVRT ・多心動週期平均（5 週期） ━━━━━━━━━━━━━━━━━━━━━━━ ▎特殊情境 ②｜MAC ・E/A <0.8：正常 ・E/A >1.8：LA 壓力升高 ・IVRT <80 ms：LA 壓力升高 ━━━━━━━━━━━━━━━━━━━━━━━ ▎舒張期運動壓力測試 ▸ 測量時機：跑步機後 1–2 分鐘恢復期（心跳約 100） 【陽性判讀】 ・Septal E/E' ≥ 15 ・Average E/E' ≥ 14 ・TR velocity >3.2 m/s（非靜息 2.8） ━━━━━━━━━━━━━━━━━━━━━━━ ▎結論演算法 ① 三大核心：E'、E/E'、TR velocity（+
